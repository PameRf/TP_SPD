#include <Keypad.h>
#include <LiquidCrystal.h>

const byte FILAS=4;
const byte COLUMNAS=4;
char caracteres[4][4]={
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'},
};
				
byte pinesFilas[]={7,6,5,4};
				
byte pinesColumnas[]={3,2,1,0};
				
Keypad miTeclado= Keypad (makeKeymap(caracteres), 
                 pinesFilas, pinesColumnas, FILAS,COLUMNAS);
void setup()
{
  Serial.begin(9600);
}

void loop()
{
  char boton= miTeclado.getKey();
  
  if(boton != NO_KEY){
    Serial.println(boton);
  }
  delay(25); 
}
